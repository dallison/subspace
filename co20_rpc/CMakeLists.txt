# CMakeLists.txt for the co20_rpc subdirectory
#
# This mirrors the Bazel build under //co20_rpc.  It builds the C++20 stackless
# coroutine (co20) RPC runtime libraries, generates the co20-flavored RPC stubs
# from rpc/proto/rpc_test.proto using the subspace_rpc plugin defined in
# rpc/CMakeLists.txt, and builds the tests and example binaries.
#
# It is added from the top-level CMakeLists.txt after add_subdirectory(rpc) so
# the subspace_rpc_plugin host tool is already defined.  Unlike asio_rpc, co20
# has no third-party dependency (it only needs co_cpp20), but every target that
# includes the co20 headers must be compiled as C++20.

cmake_minimum_required(VERSION 3.15)

# The co20 stubs are generated by the subspace_rpc plugin.  When cross-compiling
# without a host plugin, rpc/CMakeLists.txt returns early and never defines the
# plugin target; in that case skip co20_rpc too (it is not part of the
# cross-compiled deliverables, matching the Bazel build).
if(NOT TARGET subspace_rpc_plugin)
    message(STATUS
        "Skipping co20_rpc/ CMake build: subspace_rpc_plugin is not available "
        "(cross-compiling without -DSUBSPACE_RPC_PLUGIN).")
    return()
endif()

include(FetchContent)
FetchContent_GetProperties(protobuf)

# Where the co20-flavored stubs are generated.  Its layout reproduces the Bazel
# include path "co20_rpc/proto/rpc_test_co20_rpc/rpc/proto/rpc_test.subspace.*".
# The plugin keeps the proto's "rpc/proto/" prefix, so this root must hold the
# files exactly where the sources #include them.
set(CO20_RPC_GEN_ROOT ${CMAKE_BINARY_DIR}/co20_rpc/proto/rpc_test_co20_rpc)

# The rpc/ build also emits a "rpc/proto/rpc_test.subspace.rpc_*.h" under
# ${CMAKE_BINARY_DIR} (the co::Coroutine flavor), which the top-level
# include_directories(${CMAKE_BINARY_DIR}) puts on the search path ahead of any
# target-level include directory.  Prepend the co20 gen root for this whole
# subdirectory so the co20-flavored headers win over the co/ ones for the
# shared "rpc/proto/..." include path.
include_directories(BEFORE ${CO20_RPC_GEN_ROOT})

# --- Runtime libraries ---

add_library(co20_rpc_server STATIC
    server/rpc_server.cc
    server/rpc_server.h
)
target_link_libraries(co20_rpc_server PUBLIC
    subspace_common
    subspace_client
    subspace_proto
    rpc_common
    absl::flat_hash_map
    absl::flat_hash_set
    absl::status
    absl::statusor
    absl::str_format
    co_cpp20
    protobuf::libprotobuf
)

add_library(co20_rpc_client STATIC
    client/rpc_client.cc
    client/rpc_client.h
)
target_link_libraries(co20_rpc_client PUBLIC
    co20_rpc_server
    subspace_common
    subspace_client
    subspace_proto
    rpc_common
    absl::flat_hash_map
    absl::flat_hash_set
    absl::status
    absl::statusor
    absl::str_format
    co_cpp20
    protobuf::libprotobuf
)

# --- Generated co20 RPC stubs ---
set(CO20_RPC_PROTO_FILE ${CMAKE_SOURCE_DIR}/rpc/proto/rpc_test.proto)

set(CO20_RPC_CLIENT_SRC ${CO20_RPC_GEN_ROOT}/rpc/proto/rpc_test.subspace.rpc_client.cc)
set(CO20_RPC_CLIENT_HDR ${CO20_RPC_GEN_ROOT}/rpc/proto/rpc_test.subspace.rpc_client.h)
set(CO20_RPC_SERVER_SRC ${CO20_RPC_GEN_ROOT}/rpc/proto/rpc_test.subspace.rpc_server.cc)
set(CO20_RPC_SERVER_HDR ${CO20_RPC_GEN_ROOT}/rpc/proto/rpc_test.subspace.rpc_server.h)

add_custom_command(
    OUTPUT ${CO20_RPC_CLIENT_SRC} ${CO20_RPC_CLIENT_HDR}
           ${CO20_RPC_SERVER_SRC} ${CO20_RPC_SERVER_HDR}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${CO20_RPC_GEN_ROOT}
    COMMAND $<TARGET_FILE:protobuf::protoc>
    ARGS --plugin=protoc-gen-subspace_rpc=$<TARGET_FILE:subspace_rpc_plugin>
         --subspace_rpc_out=rpc_style=co20:${CO20_RPC_GEN_ROOT}
         --proto_path=${CMAKE_SOURCE_DIR}
         --proto_path=${protobuf_SOURCE_DIR}/src
         ${CO20_RPC_PROTO_FILE}
    DEPENDS ${CO20_RPC_PROTO_FILE} protobuf::protoc subspace_rpc_plugin
    WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
    COMMENT "Generating co20 subspace RPC stubs from rpc/proto/rpc_test.proto"
)

# The generated sources #include their own headers via the short "rpc/proto/"
# prefix, so CO20_RPC_GEN_ROOT must be on the include path; the test sources use
# the full "co20_rpc/proto/rpc_test_co20_rpc/..." path resolved by CMAKE_BINARY_DIR.
add_library(co20_rpc_test_subspace_rpc_client STATIC
    ${CO20_RPC_CLIENT_SRC} ${CO20_RPC_CLIENT_HDR})
target_include_directories(co20_rpc_test_subspace_rpc_client PUBLIC
    ${CMAKE_BINARY_DIR} ${CO20_RPC_GEN_ROOT})
target_link_libraries(co20_rpc_test_subspace_rpc_client PUBLIC
    rpc_test_cc_proto co20_rpc_client)

add_library(co20_rpc_test_subspace_rpc_server STATIC
    ${CO20_RPC_SERVER_SRC} ${CO20_RPC_SERVER_HDR})
target_include_directories(co20_rpc_test_subspace_rpc_server PUBLIC
    ${CMAKE_BINARY_DIR} ${CO20_RPC_GEN_ROOT})
target_link_libraries(co20_rpc_test_subspace_rpc_server PUBLIC
    rpc_test_cc_proto co20_rpc_server)

# Everything that includes the co20 headers (co/coroutine_cpp20.h is guarded on
# __cplusplus >= 202002L) must be compiled as C++20.
set_target_properties(
    co20_rpc_server
    co20_rpc_client
    co20_rpc_test_subspace_rpc_client
    co20_rpc_test_subspace_rpc_server
    co20_rpc_server_test
    co20_rpc_client_test
    co20_rpc_test
    co20_rpc_example_server
    co20_rpc_example_client
    PROPERTIES CXX_STANDARD 20 CXX_STANDARD_REQUIRED ON
)
